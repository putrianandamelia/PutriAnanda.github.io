  • ConnectionFactory: dengan Pooling Koneksi Database

    • Program ini menggunakan Factory Constructor untuk mengelola koneksi database. Jika koneksi dengan nama database yang sama sudah pernah dibuat, maka program akan menggunakan objek yang sudah ada (cache), sehingga tidak perlu membuat koneksi baru. Teknik ini disebut pooling koneksi.

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A

class ConnectionFactory {
  static final Map _pool = {};

  String database;

  ConnectionFactory._internal(this.database);

  factory ConnectionFactory(String database) {
    if (_pool.containsKey(database)) {
      return _pool[database]!;
    }

    var connection = ConnectionFactory._internal(database);
    _pool[database] = connection;
    return connection;
  }

  void connect() {
    print('Terhubung ke database $database');
  }
}

void main() {
  var conn1 = ConnectionFactory('MySQL');
  var conn2 = ConnectionFactory('MySQL');

  conn1.connect();

  print(identical(conn1, conn2));
}

  • NotificationFactory: yang Memilih Subclass Berdasarkan Platform

    • Program ini menggunakan Factory Constructor untuk memilih jenis notifikasi berdasarkan platform yang digunakan. Jika pengguna memilih Android, maka objek AndroidNotification akan dibuat. Jika memilih iOS atau Web, maka akan dibuat objek yang sesuai.

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A

abstract class Notification {
  void send();

  factory Notification(String platform) {
    switch (platform.toLowerCase()) {
      case 'android':
        return AndroidNotification();
      case 'ios':
        return IOSNotification();
      case 'web':
        return WebNotification();
      default:
        throw ArgumentError('Platform tidak ditemukan');
    }
  }
}

class AndroidNotification implements Notification {
  @override
  void send() {
    print('Mengirim notifikasi Android');
  }
}

class IOSNotification implements Notification {
  @override
  void send() {
    print('Mengirim notifikasi iOS');
  }
}

class WebNotification implements Notification {
  @override
  void send() {
    print('Mengirim notifikasi Web');
  }
}

void main() {
  var notif = Notification('android');
  notif.send();
}

  • ShapeFactory: dengan Cache untuk Shape yang Sama

    • Program ini menerapkan Factory Constructor dan cache object. Jika pengguna meminta shape yang sama lebih dari satu kali, program akan mengembalikan objek yang sudah tersimpan di cache sehingga tidak membuat objek baru. Hal ini membuat penggunaan memori lebih efisien.

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A

abstract class Shape {
  void draw();

  static final Map _cache = {};

  factory Shape(String type) {
    if (_cache.containsKey(type)) {
      return _cache[type]!;
    }

    Shape shape;

    switch (type.toLowerCase()) {
      case 'circle':
        shape = Circle();
        break;
      case 'square':
        shape = Square();
        break;
      default:
        throw ArgumentError('Shape tidak ditemukan');
    }

    _cache[type] = shape;

    return shape;
  }
}

class Circle implements Shape {
  @override
  void draw() {
    print('Menggambar Lingkaran');
  }
}

class Square implements Shape {
  @override
  void draw() {
    print('Menggambar Persegi');
  }
}

void main() {
  var shape1 = Shape('circle');
  var shape2 = Shape('circle');

  shape1.draw();

  print(identical(shape1, shape2));
}

Challenge: AnimalFactory

Program ini menggunakan Factory Constructor untuk membuat berbagai jenis hewan berdasarkan input pengguna. Program juga menerapkan validasi untuk memastikan nama hewan tidak kosong, cache untuk menyimpan objek yang sudah dibuat, dan subclass selection untuk memilih kelas hewan yang sesuai seperti Cat, Dog, atau Bird.

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A

abstract class Animal {
  String name;

  Animal(this.name);

  void sound();
}

class Cat extends Animal {
  Cat(String name) : super(name);

  @override
  void sound() {
    print('$name : Meong');
  }
}

class Dog extends Animal {
  Dog(String name) : super(name);

  @override
  void sound() {
    print('$name : Guk Guk');
  }
}

class Bird extends Animal {
  Bird(String name) : super(name);

  @override
  void sound() {
    print('$name : Cuit Cuit');
  }
}

class AnimalFactory {
  static final Map cache = {};

  static Animal createAnimal(String type, String name) {
    if (name.isEmpty) {
      throw Exception('Nama hewan tidak boleh kosong');
    }

    String key = '$type-$name';

    if (cache.containsKey(key)) {
      return cache[key]!;
    }

    Animal animal;

    if (type.toLowerCase() == 'cat') {
      animal = Cat(name);
    } else if (type.toLowerCase() == 'dog') {
      animal = Dog(name);
    } else if (type.toLowerCase() == 'bird') {
      animal = Bird(name);
    } else {
      throw Exception('Jenis hewan tidak tersedia');
    }

    cache[key] = animal;
    return animal;
  }
}

void main() {
  Animal animal1 = AnimalFactory.createAnimal('cat', 'Milo');
  Animal animal2 = AnimalFactory.createAnimal('cat', 'Milo');

  animal1.sound();

  print(identical(animal1, animal2));
}
