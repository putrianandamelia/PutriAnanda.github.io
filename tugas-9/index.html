Pengantar Extension Methods

Extension Methods adalah fitur pada Dart yang digunakan untuk menambahkan method baru ke tipe data yang sudah ada tanpa mengubah source code aslinya. Dengan extension methods, kode menjadi lebih rapi, mudah dibaca, dan lebih natural digunakan.

Masalah Tanpa Extension Methods

Pada kode ini, fungsi tambahan masih dibuat menggunakan StringUtils. Cara ini membuat penulisan kode lebih panjang karena method harus dipanggil melalui class. 

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A

class StringUtils {
  static String capitalize(String text) {
    if (text.isEmpty) return text;
    return text[0].toUpperCase() + text.substring(1);
  }

  static bool isEmail(String text) {
    return text.contains('@');
  }
}

void main() {
  var name = 'putri';

  var capitalized = StringUtils.capitalize(name);
  var emailCheck = StringUtils.isEmail('test@mail.com');

  print('Original name: $name');
  print('Capitalized name: $capitalized');
  print('Is "test@mail.com" an email? $emailCheck');
}

Solusi dengan Extension Methods

Extension methods membuat method bisa langsung digunakan pada object, seperti name.capitalize(), sehingga kode lebih rapi dan mudah dibaca. 

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A

extension StringExtensions on String {
  String capitalize() {
    if (isEmpty) return this;
    return this[0].toUpperCase() + substring(1);
  }
  
  bool get isEmail => contains('@');
  
  String reverse() {
    return split('').reversed.join();
  }
}

void main(){
var name = 'putri';
var capitalized = name.capitalize();  
var isEmail = 'test@mail.com'.isEmail; 
var reversed = 'hello'.reverse();    

print('Original name: $name');
print('Capitalized name: $capitalized');
print('Is "test@mail.com" an email? $isEmail');
print('Reversed "hello": $reversed');
  
}

Sintaks Extension Methods

Sintaks extension digunakan untuk menambahkan method, getter, dan operator baru ke tipe data tertentu tanpa mengubah class aslinya. 

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A
  
extension DateTimeExtensions on DateTime {
  
  String toFormattedString() {
    return '$day/$month/$year';
  }
  
  
  bool get isToday {
    var now = DateTime.now();
    return year == now.year && 
           month == now.month && 
           day == now.day;
  }
  
  
  DateTime operator +(Duration duration) {
    return add(duration);
  }
}

void main() {
  
  print('--- toFormattedString() ---');
  final now = DateTime.now();
  print('Current date: ${now.toFormattedString()}');

  final specificDate = DateTime(2023, 10, 26);
  print('Specific date (2023-10-26): ${specificDate.toFormattedString()}');

  print('\n--- isToday ---');
  
  final today = DateTime.now();
  final tomorrow = DateTime.now().add(Duration(days: 1));
  final yesterday = DateTime.now().subtract(Duration(days: 1));

  print('Is ${today.toFormattedString()} today? ${today.isToday}');
  print('Is ${tomorrow.toFormattedString()} today? ${tomorrow.isToday}');
  print('Is ${yesterday.toFormattedString()} today? ${yesterday.isToday}');

  print('\n--- operator + ---');
  
  final initialDate = DateTime(2024, 1, 15);
  final durationToAdd = Duration(days: 10, hours: 3);
  final futureDate = initialDate + durationToAdd;

  print('Initial date: ${initialDate.toFormattedString()}');
  print('Duration to add: ${durationToAdd.inDays} days, ${durationToAdd.inHours % 24} hours');
  print('Future date: ${futureDate.toFormattedString()}');
  print('Future date (full): $futureDate');

  final anotherDuration = Duration(days: 2);
  final evenFutureDate = futureDate + anotherDuration;
  print('Future date + 2 day: ${evenFutureDate.toFormattedString()}');
}

Extension untuk Null Safety

Extension ini membantu menangani data null agar program lebih aman dan mengurangi error saat dijalankan.

extension NullableStringExtensions on String? {
  String orEmpty() => this ?? '';
  
  bool get isNullOrEmpty => this == null || this!.isEmpty;
  
  bool get isNotNullOrEmpty => !isNullOrEmpty;
  
  String withDefault(String defaultValue) => this ?? defaultValue;
}

void main(){

Praktik 1 : Utilities untuk Dart Types

Pada kode ini ditambahkan fungsi tambahan pada tipe num, seperti format Rupiah, pengecekan range angka, dan perhitungan pangkat. 

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A
  
extension NumberExtensions on num {
  
  String toRupiah() {
    return 'Rp${toStringAsFixed(0).replaceAllMapped(
      RegExp(r'(\d{1,3})(?=(\d{3})+(?!\d))'),
      (match) => '${match[1]}.',
    )}';
  }
  
  bool isBetween(num min, num max) {
    return this >= min && this <= max;
  }
  
  num power(int exponent) {
    var result = this;
    for (var i = 1; i < exponent; i++) {
      result *= this;
    }
    return result;
  }
}

void main(){

var price = 1500000;
print(price.toRupiah());  
print(5.isBetween(1, 10)); 
print(2.power(3));        
}

Generic Extensions

Generic extension memungkinkan extension digunakan untuk berbagai tipe data sehingga lebih fleksibel dan dapat dipakai ulang.

extension ListExtensions on List {
  
  T? getOrNull(int index) {
    if (index < 0 || index >= length) return null;
    return this[index];
  }
  
  List> chunk(int size) {
    var chunks = >[];
    for (var i = 0; i < length; i += size) {
      chunks.add(sublist(i, i + size > length ? length : i + size));
    }
    return chunks;
  }
  
  List distinct() {
    var result = [];
    for (var item in this) {
      if (!result.contains(item)) {
        result.add(item);
      }
    }
    return result;
  }
}

print(numbers.chunk(3));  
print([1, 2, 2, 3, 3, 3].distinct());

Extension untuk Custom Classes

Extension digunakan untuk menambahkan fitur baru pada class Product, seperti diskon, format harga, dan pengecekan harga mahal. 

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A
  
class Product {
  final String name;
  final double price;
  
  Product(this.name, this.price);
}

extension DoubleRupiahFormatter on double {
  String toRupiah() {
   
    final intValue = toInt();
    
    String numberString = intValue.toString();
    
    String formattedString = '';
    int counter = 0;
    for (int i = numberString.length - 1; i >= 0; i--) {
      formattedString = numberString[i] + formattedString;
      counter++;
      if (counter % 3 == 0 && i != 0) {
        formattedString = '.' + formattedString;
      }
    }
    
    return 'Rp$formattedString';
  }
}

extension ProductExtensions on Product {
  
  Product withDiscount(double percentage) {
    var discountedPrice = price * (1 - percentage / 100);
    return Product(name, discountedPrice);
  }
  
  String get displayInfo => '$name - ${price.toRupiah()}';
  
  bool get isExpensive => price > 1000000;
}

void main(){
  
  var laptop = Product('Handphone', 15000000);
  var discounted = laptop.withDiscount(10);
  print(discounted.displayInfo);  
  print(laptop.isExpensive);      

  var keyboard = Product('Ipad', 750000);
  print(keyboard.displayInfo);
  print(keyboard.isExpensive);
}

Praktik 2 : Validator Extensions

Validator extension digunakan untuk memeriksa validitas data seperti email, password, nomor telepon, dan angka agar data lebih aman digunakan. 

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A
  
extension StringValidators on String {
  bool get isEmail {
    return RegExp(r'^[\w-\.]+@([\w-]+\.)+[\w-]{2,4}$').hasMatch(this);
  }

  bool get isPhoneNumber {
    return RegExp(r'^[0-9]{10,13}$').hasMatch(this);
  }

  bool get isStrongPassword {
    return length >= 8 &&
        contains(RegExp(r'[A-Z]')) &&
        contains(RegExp(r'[a-z]')) &&
        contains(RegExp(r'[0-9]'));
  }

  bool get isUrl {
    try {
      Uri.parse(this);
      return true;
    } catch (_) {
      return false;
    }
  }
}

extension IntValidators on int {
  bool get isAdultAge => this >= 18;
  bool get isPositive => this > 0;
  bool get isEven => this % 2 == 0;
  bool get isOdd => this % 2 != 0;
}

void main() {
  
  var email = 'putri14@gmail.com';
  var age = 20;
  var password = 'putri0412';

  print(email.isEmail); 
  print(age.isAdultAge); 
  print(password.isStrongPassword); 
  print(10.isEven); 
}

Ringkasan & Best practice

Extension methods digunakan untuk menambahkan method baru pada tipe data tanpa mengubah class asli. Fitur ini membuat kode lebih rapi, singkat, dan mudah dibaca.

Best practices dalam penggunaan extension methods:

  • Gunakan nama extension yang jelas
  • Pisahkan extension sesuai tipe data
  • Jangan menggunakan extension secara berlebihan
  • Perhatikan null safety
  • Tambahkan dokumentasi pada extension penting
