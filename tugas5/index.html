Multiple Inheritance dengan Mixin

Multiple inheritance: digunakan agar satu objek mewarisi perilaku/properti dari dua atau lebih class sekaligus. Dart: pakai mixin untuk menggabungkan fungsionalitas beberapa class tanpa pewarisan rumit. Bahasa yang mendukung: C++, Python, Scala, Perl (multiple inheritance langsung), Java, C#, Kotlin (single inheritance). Contoh: Bebek (berenang + terbang), Smartphone (kamera + telepon).

    print('Can swim!');

class Duck extends Bird with CanSwim {}

  Bird bird = Bird();
  bird.fly();

  Duck duck = Duck();
  duck.fly();
  duck.swim();

mixin CanFly {
  bool _isFlying = false;

  bool get isFlying => _isFlying;

  void fly() {
    if (!_isFlying) {
      _isFlying = true;
      print('Started flying!');
    } else {
      print('Already flying!');
    }
  }

  void land() {
    if (_isFlying) {
      _isFlying = false;
      print('Landed!');
    } else {
      print('Already on the ground!');
    }
  }
}

mixin CanSwim {
  bool _isSwimming = false;

  bool get isSwimming => _isSwimming;

  void swim() {
    if (!_isSwimming) {
      _isSwimming = true;
      print('Started swimming!');
    } else {
      print('Already swimming!');
    }
  }

  void stopSwim() {
    if (_isSwimming) {
      _isSwimming = false;
      print('Stop swimming!');
    } else {
      print('Already stopped swimming!');
    }
  }
}

class Bat with CanFly {}

class Frog with CanSwim {}

class FlyingFish with CanFly, CanSwim {}

class Dolphin with CanSwim {}

void main() {
  Bat bat = Bat();
  bat.fly();
  bat.land();

  print('Is Bat flying? : ${bat.isFlying}');

  print('----------------');

  Frog frog = Frog();
  frog.swim();
  frog.stopSwim();

  print('Is Frog swimming? : ${frog.isSwimming}');

  print('----------------');

  FlyingFish fish = FlyingFish();
  fish.swim();
  fish.fly();
  fish.land();
  fish.stopSwim();

  print('Is FlyingFish swimming? : ${fish.isSwimming}');

  print('----------------');

  Dolphin dolphin = Dolphin();
  dolphin.swim();
  dolphin.stopSwim();
}

Sealed Class

Exhaustive checking: Compiler tahu semua kemungkinan subclass, Pattern matching: Switch case yang aman, Documentation: Jelas hierarki terbatas. 

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A

sealed class PaymentResponse {}

class PaymentSuccess extends PaymentResponse {
  final String message;
  PaymentSuccess(this.message);
}

class PaymentError extends PaymentResponse {
  final String errorMessage;
  PaymentError(this.errorMessage);
}

void handleResponse(PaymentResponse response) {
  if (response is PaymentSuccess) {
    print("Payment Success: ${response.message}");
  } else if (response is PaymentError) {
    print("Payment Error: ${response.errorMessage}");
  }
}

void main() {
  handleResponse(PaymentSuccess("Pembayaran berhasil"));
  handleResponse(PaymentError("Saldo tidak cukup"));
}

Final Class

Tidak bisa di-extend atau di-implement dari luar library, Tapi bisa di-extend/implement di dalam library yang sama, Berguna untuk class yang ingin "ditutup" untuk modifikasi.

final class Settings {

  final int fontSize;

  Settings({required this.theme, required this.fontSize});

    print("Theme: $theme, Font Size: $fontSize");
  }
}

Praktik 1 : Sealed Class dengan Pattern Matching

// Nama  : Putri Ananda Amelia
// NIM   : 251410029
// Kelas : SI2A
  
sealed class LoginStatus {
  const LoginStatus();
}

class LoggingIn extends LoginStatus {
  const LoggingIn();
}

class Success extends LoginStatus {
  final String username;
  const Success(this.username);
}

class Error extends LoginStatus {
  final String message;
  const Error(this.message);
}

void handleStatus(LoginStatus status) {
  switch (status) {
    case LoggingIn():
      print('Logging in...');
    case Success(:final username):
      print('Welcome, $username');
    case Error(:final message):
      print('Login failed: $message');
  }
}

void main() {
  final states = [
    LoggingIn(),
    Success('Putri'),
    Error('Wrong password'),
  ];

  for (var state in states) {
    handleStatus(state);
  }
}

sealed class DataState {}

class LoadingData extends DataState {}

class DataLoaded extends DataState {
  final List data;
  DataLoaded(this.data);
}

class DataError extends DataState {
  final String message;
  DataError(this.message);
}

void updateUI(DataState state) {
  switch (state) {
    case LoadingData():
      showLoading();
    case DataLoaded(:final data):
      showData(data);
    case DataError(:final message):
      showError(message);
  }
}

void showLoading() {
  print('⏳ Loading data...');
}

void showData(List data) {
  print('✅ Data loaded:');
  for (int i = 0; i < data.length; i++) {
    print('   ${i + 1}. ${data[i]}');
  }
}

void showError(String message) {
  print('❌ Error: $message');
}

void main() {
  final states = [
    LoadingData(),
    DataLoaded(['Data 1', 'Data 2']),
    DataError('Gagal memuat data'),
  ];

  for (var state in states) {
    updateUI(state);
  }
}

Kesimpulan

Sealed class cocok untuk membuat tipe data dengan hierarki terbatas agar pattern matching aman. Base class digunakan untuk membuat class dasar yang bisa diturunkan di framework atau aplikasi. Final class cocok untuk utility class yang tidak boleh diturunkan atau diimplementasi dari luar.

final class MathUtils {
 
  static int tambah(int a, int b) {
    return a + b;
  }

  static int kurang(int a, int b) {
    return a - b;
  }

  static int kali(int a, int b) {
    return a * b;
  }

  static double bagi(int a, int b) {
    return a / b;
  }
}

  repo.read();

  print("Tambah: ${MathUtils.tambah(10, 5)}");
